-- Luyện tập về JOIN nhiều bảng

-- a. MaHS, HoTenHS, GioiTinh, MaLop, TenLop, NamHoc, MaGVCN, HoTenGV (chủ nhiệm) của học sinh trong trường.
select 
    hoc_sinh.ma_hs, 
    hoc_sinh.ho_ten_hs, 
    hoc_sinh.gioi_tinh, 
    hoc_sinh.ma_lop, 
    lop.ten_lop, 
    lop.nam_hoc, 
    lop.ma_gvcn, 
    giao_vien.ho_ten_gv
from hoc_sinh
join lop on hoc_sinh.ma_lop = lop.ma_lop 
join giao_vien on lop.ma_gvcn = giao_vien.ma_gv;

-- b. MaHS, HoTenHS, HocKy, MaMH, TenMH, DiemThiGiuaKy, DiemThiCuoiKy của học sinh và các môn học đã có kết quả tương ứng với từng học sinh trong trường.
select 
	hoc_sinh.ma_hs,
    hoc_sinh.ho_ten_hs,
	ket_qua_hoc_tap.hoc_ky, 
    ket_qua_hoc_tap.ma_mh,
	mon_hoc.ten_mh,
    ket_qua_hoc_tap.diem_thi_giua_ky, 
    ket_qua_hoc_tap.diem_thi_cuoi_ky
from ket_qua_hoc_tap
join hoc_sinh on ket_qua_hoc_tap.ma_hs = hoc_sinh.ma_hs
join mon_hoc on ket_qua_hoc_tap.ma_mh = mon_hoc.ma_mh;

-- c. MaHS, HoTenHS, HocKy, MaMH, TenMH, DiemThiGiuaKy, DiemThiCuoiKy, MaLop, MaGV (phụ trách), HoTenGV (phụ trách) của học sinh và các môn học đã có kết quả tương ứng với từng học sinh trong trường.
select 
	hoc_sinh.ma_hs,
	hoc_sinh.ho_ten_hs,
    ket_qua_hoc_tap.hoc_ky,
    ket_qua_hoc_tap.ma_mh, 
    mon_hoc.ten_mh,
    ket_qua_hoc_tap.diem_thi_giua_ky,
    ket_qua_hoc_tap.diem_thi_cuoi_ky,
    hoc_sinh.ma_lop,
    lop.ma_gvcn, 
    giao_vien.ho_ten_gv

from ket_qua_hoc_tap
join hoc_sinh on ket_qua_hoc_tap.ma_hs = hoc_sinh.ma_hs
join mon_hoc on ket_qua_hoc_tap.ma_mh = mon_hoc.ma_mh
join lop on hoc_sinh.ma_lop = lop.ma_lop
join giao_vien on lop.ma_gvcn = giao_vien.ma_gv;

-- Câu d
select 
    ket_qua_hoc_tap.ma_hs, 
    hoc_sinh.ho_ten_hs, 
    hoc_sinh.ma_lop, 
    lop.ma_gvcn, 
    giao_vien_chu_nhiem.ho_ten_gv as ho_ten_gv_chu_nhiem,
    ket_qua_hoc_tap.hoc_ky, 
    ket_qua_hoc_tap.ma_mh, 
    mon_hoc.ten_mh, 
    ket_qua_hoc_tap.diem_thi_giua_ky, 
    ket_qua_hoc_tap.diem_thi_cuoi_ky,
    lop.ma_gvcn as ma_gv_phu_trach, 
    giao_vien_phu_trach.ho_ten_gv as ho_ten_gv_phu_trach
from ket_qua_hoc_tap
join hoc_sinh on ket_qua_hoc_tap.ma_hs = hoc_sinh.ma_hs
join lop on hoc_sinh.ma_lop = lop.ma_lop
join giao_vien as giao_vien_chu_nhiem on lop.ma_gvcn = giao_vien_chu_nhiem.ma_gv
join mon_hoc on ket_qua_hoc_tap.ma_mh = mon_hoc.ma_mh
left join giao_vien as giao_vien_phu_trach on lop.ma_gvcn = giao_vien_phu_trach.ma_gv;

-- e. MaHS, HoTenHS, MaLop, MaGVCN, HoTenGV (chủ nhiệm), HocKy, MaMH, TenMH, DiemThiGiuaKy, DiemThiCuoiKy, MaGV (phụ trách), HoTenGV (phụ trách) 
-- của những học sinh nữ với các môn học đã có kết quả thi cuối kỳ hoặc giữa kỳ được 9 điểm trở lên.

select 
    ket_qua_hoc_tap.ma_hs, 
    hoc_sinh.ho_ten_hs, 
    hoc_sinh.ma_lop, 
    lop.ma_gvcn, 
    giao_vien_chu_nhiem.ho_ten_gv as ho_ten_gv_chu_nhiem,
    ket_qua_hoc_tap.hoc_ky, 
    ket_qua_hoc_tap.ma_mh, 
    mon_hoc.ten_mh, 
    ket_qua_hoc_tap.diem_thi_giua_ky, 
    ket_qua_hoc_tap.diem_thi_cuoi_ky,
    lop.ma_gvcn as ma_gv_phu_trach, 
    giao_vien_phu_trach.ho_ten_gv as ho_ten_gv_phu_trach
from ket_qua_hoc_tap
join hoc_sinh on ket_qua_hoc_tap.ma_hs = hoc_sinh.ma_hs
join lop on hoc_sinh.ma_lop = lop.ma_lop
join giao_vien as giao_vien_chu_nhiem on lop.ma_gvcn = giao_vien_chu_nhiem.ma_gv
join mon_hoc on ket_qua_hoc_tap.ma_mh = mon_hoc.ma_mh
left join giao_vien as giao_vien_phu_trach on lop.ma_gvcn = giao_vien_phu_trach.ma_gv
where (ket_qua_hoc_tap.diem_thi_cuoi_ky >= 9 or ket_qua_hoc_tap.diem_thi_giua_ky >= 9) and hoc_sinh.gioi_tinh = 'nữ';


-- f. MaHS, HoTenHS, HocKy, MaMH, TenMH, DiemThiGiuaKy, DiemThiCuoiKy, MaLop, MaGV (phụ trách), HoTenGV (phụ trách) 
-- của học sinh và các môn học đã có kết quả tương ứng với từng học sinh trong trường. Với điều kiện là chỉ hiển thị những 
-- môn học mà giáo viên phụ trách môn đó cũng chính là giáo viên chủ nhiệm của lớp.

select 
    ket_qua_hoc_tap.ma_hs, 
    hoc_sinh.ho_ten_hs, 
    ket_qua_hoc_tap.hoc_ky, 
    ket_qua_hoc_tap.ma_mh, 
    mon_hoc.ten_mh, 
    ket_qua_hoc_tap.diem_thi_giua_ky, 
    ket_qua_hoc_tap.diem_thi_cuoi_ky,
    hoc_sinh.ma_lop, 
    lop.ma_gvcn, 
    giao_vien_chu_nhiem.ho_ten_gv as ho_ten_gv_chu_nhiem
from ket_qua_hoc_tap
join hoc_sinh on ket_qua_hoc_tap.ma_hs = hoc_sinh.ma_hs
join lop on hoc_sinh.ma_lop = lop.ma_lop
join giao_vien as giao_vien_chu_nhiem on lop.ma_gvcn = giao_vien_chu_nhiem.ma_gv
join mon_hoc on ket_qua_hoc_tap.ma_mh = mon_hoc.ma_mh
left join phu_trach_bo_mon on lop.ma_gvcn = phu_trach_bo_mon.ma_gvpt and mon_hoc.ma_mh = phu_trach_bo_mon.ma_mh and ket_qua_hoc_tap.hoc_ky = phu_trach_bo_mon.hoc_ky
where phu_trach_bo_mon.ma_gvpt is not null;

-- g. MaMH, MaLop, HocKy của những môn học đã được thi cuối kỳ vào năm 2023
select 
    ket_qua_hoc_tap.ma_mh, 
    lop.ma_lop, 
    ket_qua_hoc_tap.hoc_ky
from ket_qua_hoc_tap
join hoc_sinh on ket_qua_hoc_tap.ma_hs = hoc_sinh.ma_hs
join lop on hoc_sinh.ma_lop = lop.ma_lop
where year(ket_qua_hoc_tap.ngay_gio_thi_cuoi_ky) = 2023;

-- h. MaMH, MaLop, HocKy của những môn học đã được thi vào tháng 8 năm 2023
select 
    ket_qua_hoc_tap.ma_mh, 
    lop.ma_lop, 
    ket_qua_hoc_tap.hoc_ky
from ket_qua_hoc_tap
join hoc_sinh on ket_qua_hoc_tap.ma_hs = hoc_sinh.ma_hs
join lop on hoc_sinh.ma_lop = lop.ma_lop
where year(ket_qua_hoc_tap.ngay_gio_thi_cuoi_ky) = 2023 and month(ket_qua_hoc_tap.ngay_gio_thi_cuoi_ky) = 8;

-- i. MaMH, MaLop, HocKy của những môn học đã được thi trước ngày 20 tháng 8 năm 2023
select 
    ket_qua_hoc_tap.ma_mh, 
    lop.ma_lop, 
    ket_qua_hoc_tap.hoc_ky
from ket_qua_hoc_tap
join hoc_sinh on ket_qua_hoc_tap.ma_hs = hoc_sinh.ma_hs
join lop on hoc_sinh.ma_lop = lop.ma_lop
where ket_qua_hoc_tap.ngay_gio_thi_cuoi_ky < '2023-08-20';

-- k. MaMH, MaLop, HocKy của những môn học đã được thi sau ngày 20 tháng 8 năm 2023 đúng 21 ngày.
select 
    ket_qua_hoc_tap.ma_mh, 
    lop.ma_lop, 
    ket_qua_hoc_tap.hoc_ky
from ket_qua_hoc_tap
join hoc_sinh on ket_qua_hoc_tap.ma_hs = hoc_sinh.ma_hs
join lop on hoc_sinh.ma_lop = lop.ma_lop
where ket_qua_hoc_tap.ngay_gio_thi_cuoi_ky >= '2023-09-10' and ket_qua_hoc_tap.ngay_gio_thi_cuoi_ky <= '2023-10-01';

-- l. MaMH, MaLop, HocKy của những môn học đã được thi trong khoảng từ ngày 10 đến ngày 20 tháng 8 năm 2023.
select 
    ket_qua_hoc_tap.ma_mh, 
    lop.ma_lop, 
    ket_qua_hoc_tap.hoc_ky
from ket_qua_hoc_tap
join hoc_sinh on ket_qua_hoc_tap.ma_hs = hoc_sinh.ma_hs
join lop on hoc_sinh.ma_lop = lop.ma_lop
where ket_qua_hoc_tap.ngay_gio_thi_cuoi_ky >= '2023-08-10' and ket_qua_hoc_tap.ngay_gio_thi_cuoi_ky <= '2023-08-20';

-- câu m
-- Trong câu truy vấn này, chúng ta sử dụng điều kiện kq.ngay_gio_thi_cuoi_ky >= '2023-08-10' AND kq.ngay_gio_thi_cuoi_ky <= '2023-08-20' 
-- để lấy ra những môn học được thi trong khoảng từ ngày 10 đến ngày 20 tháng 8 năm 2023.
-- m. MaMH, MaLop, HocKy của những môn học đã được thi trong khoảng từ 10 giờ 00 phút ngày 10 đến 20 giờ 30 phút ngày 20 tháng 8 năm 2023.

select 
    ket_qua_hoc_tap.ma_mh, 
    lop.ma_lop, 
    ket_qua_hoc_tap.hoc_ky
from ket_qua_hoc_tap
join hoc_sinh on ket_qua_hoc_tap.ma_hs = hoc_sinh.ma_hs
join lop on hoc_sinh.ma_lop = lop.ma_lop
where ket_qua_hoc_tap.ngay_gio_thi_cuoi_ky >= '2023-08-20 10:00:00' and ket_qua_hoc_tap.ngay_gio_thi_cuoi_ky <= '2023-08-20 20:30:00';
